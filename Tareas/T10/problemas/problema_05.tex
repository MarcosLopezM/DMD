\begin{problema}[2]
	Estimar el espesor de la capa límite al \(\qty{99}{\percent}\) de los
	siguientes casos:

	\begin{enumerate}
		\item El ala de un avión de papel: \(\qty{0.25}{\m}\), velocidad
		      \(\qty{1}{\m\per\s}\)
		\item La parte inferior de un supertanker: longitud \(\qty{300}{\m}\),
		      velocidad \(\qty{5}{\m\per\s}\)
		\item Una pista de aeropuerto en un día ventoso: longitud \(\qty{5}{\km}\),
		      velocidad \(\qty{10}{\m\per\s}\)
		\item ¿Serán exactas estas estimaciones en cada paso? ¿Por qué?
	\end{enumerate}
\end{problema}
